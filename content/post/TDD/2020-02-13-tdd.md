const solution = (clothes) => {
  const clothesKind = [];

  clothes.forEach(([_, v]) => clothesKind.push(v));
  clothesKind.sort();

  let tempt = clothesKind.filter(v => v == clothesKind[0]).length;
  for (let i = 0; i < tempt; i++ ) {
    clothesKind.shift();
  }

  console.log(clothesKind)
};
