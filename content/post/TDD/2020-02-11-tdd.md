```javascript
const solution = (heights) => {
  const answer = [];

  for (let i = 0; i < heights.length; i++) {
    answer.push(leftBigNum(heights, i));
  }

  return answer;
};

const leftBigNum = (heights, i) => {
  for (let j = i - 1; j >= 0; j--) {
    if (heights[i] < heights[j]) {
      return j + 1;
    }
    return 0;
  }
};
```
